з»„еҗҲ(Composition)APIвҖӢ

Vueе®ҳж–№и®ӨдёәVue2зҡ„з»„д»¶пјҢеҪ“жңүеӨ§йҮҸзҡ„methodsпјҢпјҢеӨ§йҮҸзҡ„watchпјҢеӨ§йҮҸзҡ„computedпјҢdataйҮҢйқўзҡ„ж•°жҚ®иҝҮеӨҡзҡ„ж—¶еҖҷпјҢзңӢд»Јз ҒдёҚеҘҪзңӢпјҢиҝҮдәҺзўҺзүҮеҢ–пјҢдәҺжҳҜVueе®ҳж–№е°ұжҸҗжЎҲдәҶVue3зҡ„з»„еҗҲејҸAPIгҖӮ з»„еҗҲејҸдёәдәҶеўһеҠ еҶ…иҒҡпјҢеҮҸе°‘зўҺзүҮеҢ–иҖҢеӯҳеңЁгҖӮдёҖеҲҮзҡ„иө·зӮ№жҳҜsetupгҖӮ

е®ҳзҪ‘еңЁз»„еҗҲејҸAPIеј•еҮәзҡ„ж–№ејҸпјҢhttps://v3.cn.vuejs.org/guide/composition-api-introduction.html

з»„еҗҲејҸе’ҢReactзҡ„HookжҖқжғіиҝ‘дјјпјҢдҪҶжҳҜеҸҲдёҚе®Ңе…ЁдёҖж ·гҖӮ

setupж ёеҝғвҖӢ

  1. Vue3дёӯпјҢж–°еўһдәҶдёҖдёӘеҮҪж•°пјҢеҗҚеӯ—еҸ«еҒҡsetupгҖӮ setupжҳҜдёҖеҲҮз»„еҗҲејҸAPIзҡ„еҹәеә§гҖӮ з»„д»¶зҡ„ж•°жҚ®е’Ңж–№жі•еқҮйңҖиҰҒеҚёиҪҪsetupеҮҪж•°дёӯгҖӮ
  2. иҝ”еӣһеҖјжңүдёӨз§Қпјҡ
  • еҜ№иұЎпјҢеҜ№иұЎдёӯзҡ„еұһжҖ§гҖҒж–№жі•, еңЁжЁЎжқҝдёӯеқҮеҸҜд»ҘзӣҙжҺҘдҪҝз”ЁгҖӮпјҲеёёз”Ёпјү
  • hеҢ…иЈ№зҡ„жёІжҹ“еҮҪж•°пјҢдҪңз”ЁжҳҜиҮӘе®ҡд№үжёІжҹ“еҶ…е®№гҖӮпјҲдёҚеёёз”ЁпјҢдәҶи§Јпјү

  1. жіЁж„ҸзӮ№пјҡ

    • е°ҪйҮҸдёҚиҰҒдёҺVue2.xй…ҚзҪ®ж··з”Ё
      • Vue2.xй…ҚзҪ®пјҲdataгҖҒmethodsгҖҒcomputed...пјүдёӯеҸҜд»Ҙи®ҝй—®еҲ°setupдёӯзҡ„еұһжҖ§гҖҒж–№жі•гҖӮ
      • дҪҶеңЁsetupдёӯдёҚиғҪи®ҝй—®еҲ°Vue2.xй…ҚзҪ®пјҲdataгҖҒmethodsгҖҒcomputed...пјүгҖӮ
      • еҰӮжһңжңүйҮҚеҗҚ, setupдјҳе…ҲгҖӮ
    • setupдёҚиғҪжҳҜдёҖдёӘasyncеҮҪж•°пјҢеӣ дёәиҝ”еӣһеҖјдёҚеҶҚжҳҜreturnзҡ„еҜ№иұЎ, иҖҢжҳҜpromise, жЁЎжқҝзңӢдёҚеҲ°returnеҜ№иұЎдёӯзҡ„еұһжҖ§гҖӮ

    пјҲеҗҺжңҹд№ҹеҸҜд»Ҙиҝ”еӣһдёҖдёӘPromiseе®һдҫӢпјҢдҪҶйңҖиҰҒSuspenseе’ҢејӮжӯҘз»„д»¶зҡ„й…ҚеҗҲпјү

refеҮҪж•°вҖӢ

з”ЁдәҺе®ҡд№үдёҖдёӘе“Қеә”ејҸзҡ„ж•°жҚ® з”Ёжі• const xxx = ref(initValue)

  • еҲӣе»әдёҖдёӘеҢ…еҗ«е“Қеә”ејҸж•°жҚ®зҡ„еј•з”ЁеҜ№иұЎпјҲreferenceеҜ№иұЎпјҢз®Җз§°refеҜ№иұЎпјүгҖӮ
  • JSдёӯж“ҚдҪңж•°жҚ®пјҡ xxx.value
  • жЁЎжқҝдёӯиҜ»еҸ–ж•°жҚ®: дёҚйңҖиҰҒ.valueпјҢзӣҙжҺҘпјҡ<div>{{xxx}}</div>
  • еӨҮжіЁпјҡ
    • жҺҘж”¶зҡ„ж•°жҚ®еҸҜд»ҘжҳҜпјҡеҹәжң¬зұ»еһӢгҖҒд№ҹеҸҜд»ҘжҳҜеҜ№иұЎзұ»еһӢгҖӮ
    • еҹәжң¬зұ»еһӢзҡ„ж•°жҚ®пјҡе“Қеә”ејҸдҫқз„¶жҳҜйқ Object.defineProperty()зҡ„getдёҺsetе®ҢжҲҗзҡ„гҖӮ
    • еҜ№иұЎзұ»еһӢзҡ„ж•°жҚ®пјҡеҶ…йғЁжұӮеҠ©дәҶVue3.0дёӯзҡ„дёҖдёӘж–°еҮҪж•°вҖ”вҖ” reactiveеҮҪж•°гҖӮ

reactiveеҮҪж•°вҖӢ

  • дҪңз”Ё: е®ҡд№үдёҖдёӘеҜ№иұЎзұ»еһӢзҡ„е“Қеә”ејҸж•°жҚ®пјҲеҹәжң¬зұ»еһӢдёҚиҰҒз”Ёе®ғпјҢиҰҒз”ЁrefеҮҪж•°пјү
  • иҜӯжі•пјҡconst д»ЈзҗҶеҜ№иұЎ = reactive(жәҗеҜ№иұЎ)жҺҘж”¶дёҖдёӘеҜ№иұЎпјҲжҲ–ж•°з»„пјүпјҢиҝ”еӣһдёҖдёӘд»ЈзҗҶеҜ№иұЎпјҲProxyзҡ„е®һдҫӢеҜ№иұЎпјҢз®Җз§°proxyеҜ№иұЎпјү
  • reactiveе®ҡд№үзҡ„е“Қеә”ејҸж•°жҚ®жҳҜж·ұеұӮж¬Ўзҡ„гҖӮ
  • еҶ…йғЁеҹәдәҺ ES6 зҡ„ Proxy е®һзҺ°пјҢйҖҡиҝҮд»ЈзҗҶеҜ№иұЎж“ҚдҪңжәҗеҜ№иұЎеҶ…йғЁж•°жҚ®иҝӣиЎҢж“ҚдҪңгҖӮ

и®Ўз®—еұһжҖ§дёҺзӣ‘и§ҶвҖӢ

computedеҮҪж•°вҖӢ

  • дёҺVue2.xдёӯcomputedй…ҚзҪ®еҠҹиғҪдёҖиҮҙ

  • еҶҷжі•

<template>
    <h1>дёҖдёӘдәәзҡ„дҝЎжҒҜ</h1>
    е§“пјҡ<input type="text" v-model="person.firstName">
    <br>
    еҗҚпјҡ<input type="text" v-model="person.lastName">
    <br>
    <span>е…ЁеҗҚпјҡ{{person.fullName}}</span>
    <br>
    е…ЁеҗҚпјҡ<input type="text" v-model="person.fullName">
</template>

<script>
    import {reactive,computed} from 'vue'
    export default {
        name: 'Demo',
        setup(){
            //ж•°жҚ®
            let person = reactive({
                firstName:'еј ',
                lastName:'дёү'
            })
            //и®Ўз®—еұһжҖ§вҖ”вҖ”з®ҖеҶҷпјҲжІЎжңүиҖғиҷ‘и®Ўз®—еұһжҖ§иў«дҝ®ж”№зҡ„жғ…еҶөпјү
            /* person.fullName = computed(()=>{
                return person.firstName + '-' + person.lastName
            }) */

            //и®Ўз®—еұһжҖ§вҖ”вҖ”е®Ңж•ҙеҶҷжі•пјҲиҖғиҷ‘иҜ»е’ҢеҶҷпјү
            person.fullName = computed({
                get(){
                    return person.firstName + '-' + person.lastName
                },
                set(value){
                    const nameArr = value.split('-')
                    person.firstName = nameArr[0]
                    person.lastName = nameArr[1]
                }
            })

            //иҝ”еӣһдёҖдёӘеҜ№иұЎпјҲеёёз”Ёпјү
            return {
                person
            }
        }
    }
</script>

watchеҮҪж•°вҖӢ

  • дёҺVue2.xдёӯwatchй…ҚзҪ®еҠҹиғҪдёҖиҮҙ

  • дёӨдёӘе°ҸвҖңеқ‘вҖқпјҡ

    • зӣ‘и§Ҷreactiveе®ҡд№үзҡ„е“Қеә”ејҸж•°жҚ®ж—¶пјҡoldValueж— жі•жӯЈзЎ®иҺ·еҸ–гҖҒејәеҲ¶ејҖеҗҜдәҶж·ұеәҰзӣ‘и§ҶпјҲdeepй…ҚзҪ®еӨұж•ҲпјүгҖӮ
    • зӣ‘и§Ҷreactiveе®ҡд№үзҡ„е“Қеә”ејҸж•°жҚ®дёӯжҹҗдёӘеұһжҖ§ж—¶пјҡdeepй…ҚзҪ®жңүж•ҲгҖӮ
    //жғ…еҶөдёҖпјҡзӣ‘и§Ҷrefе®ҡд№үзҡ„е“Қеә”ејҸж•°жҚ®
    watch(sum,(newValue,oldValue)=>{
    console.log('sumеҸҳеҢ–дәҶ',newValue,oldValue)
    },{immediate:true})
    
    //жғ…еҶөдәҢпјҡзӣ‘и§ҶеӨҡдёӘrefе®ҡд№үзҡ„е“Қеә”ејҸж•°жҚ®
    watch([sum,msg],(newValue,oldValue)=>{
    console.log('sumжҲ–msgеҸҳеҢ–дәҶ',newValue,oldValue)
    }) 
    
    /* жғ…еҶөдёүпјҡзӣ‘и§Ҷreactiveе®ҡд№үзҡ„е“Қеә”ејҸж•°жҚ®
            иӢҘwatchзӣ‘и§Ҷзҡ„жҳҜreactiveе®ҡд№үзҡ„е“Қеә”ејҸж•°жҚ®пјҢеҲҷж— жі•жӯЈзЎ®иҺ·еҫ—oldValueпјҒпјҒ
            иӢҘwatchзӣ‘и§Ҷзҡ„жҳҜreactiveе®ҡд№үзҡ„е“Қеә”ејҸж•°жҚ®пјҢеҲҷејәеҲ¶ејҖеҗҜдәҶж·ұеәҰзӣ‘и§Ҷ 
    */
    watch(person,(newValue,oldValue)=>{
    console.log('personеҸҳеҢ–дәҶ',newValue,oldValue)
    },{immediate:true,deep:false}) //жӯӨеӨ„зҡ„deepй…ҚзҪ®дёҚеҶҚеҘҸж•Ҳ
    
    //жғ…еҶөеӣӣпјҡзӣ‘и§Ҷreactiveе®ҡд№үзҡ„е“Қеә”ејҸж•°жҚ®дёӯзҡ„жҹҗдёӘеұһжҖ§
    watch(()=>person.job,(newValue,oldValue)=>{
    console.log('personзҡ„jobеҸҳеҢ–дәҶ',newValue,oldValue)
    },{immediate:true,deep:true}) 
    
    //жғ…еҶөдә”пјҡзӣ‘и§Ҷreactiveе®ҡд№үзҡ„е“Қеә”ејҸж•°жҚ®дёӯзҡ„жҹҗдәӣеұһжҖ§
    watch([()=>person.job,()=>person.name],(newValue,oldValue)=>{
    console.log('personзҡ„jobеҸҳеҢ–дәҶ',newValue,oldValue)
    },{immediate:true,deep:true})
    
    //зү№ж®Ҡжғ…еҶө
    watch(()=>person.job,(newValue,oldValue)=>{
      console.log('personзҡ„jobеҸҳеҢ–дәҶ',newValue,oldValue)
    },{deep:true}) //жӯӨеӨ„з”ұдәҺзӣ‘и§Ҷзҡ„жҳҜreactiveзҙ е®ҡд№үзҡ„еҜ№иұЎдёӯзҡ„жҹҗдёӘеұһжҖ§пјҢжүҖд»Ҙdeepй…ҚзҪ®жңүж•Ҳ

watchEffectеҮҪж•°вҖӢ

  • watchзҡ„еҘ—и·ҜжҳҜпјҡж—ўиҰҒжҢҮжҳҺзӣ‘и§Ҷзҡ„еұһжҖ§пјҢд№ҹиҰҒжҢҮжҳҺзӣ‘и§Ҷзҡ„еӣһи°ғгҖӮ

  • watchEffectзҡ„еҘ—и·ҜжҳҜпјҡдёҚз”ЁжҢҮжҳҺзӣ‘и§Ҷе“ӘдёӘеұһжҖ§пјҢзӣ‘и§Ҷзҡ„еӣһи°ғдёӯз”ЁеҲ°е“ӘдёӘеұһжҖ§пјҢйӮЈе°ұзӣ‘и§Ҷе“ӘдёӘеұһжҖ§гҖӮ

  • watchEffectжңүзӮ№еғҸcomputedпјҡ

    • дҪҶcomputedжіЁйҮҚзҡ„и®Ўз®—еҮәжқҘзҡ„еҖјпјҲеӣһи°ғеҮҪж•°зҡ„иҝ”еӣһеҖјпјүпјҢжүҖд»Ҙеҝ…йЎ»иҰҒеҶҷиҝ”еӣһеҖјгҖӮ
    • иҖҢwatchEffectжӣҙжіЁйҮҚзҡ„жҳҜиҝҮзЁӢпјҲеӣһи°ғеҮҪж•°зҡ„еҮҪж•°дҪ“пјүпјҢжүҖд»ҘдёҚз”ЁеҶҷиҝ”еӣһеҖјгҖӮ
    //watchEffectжүҖжҢҮе®ҡзҡ„еӣһи°ғдёӯз”ЁеҲ°зҡ„ж•°жҚ®еҸӘиҰҒеҸ‘з”ҹеҸҳеҢ–пјҢеҲҷзӣҙжҺҘйҮҚж–°жү§иЎҢеӣһи°ғгҖӮ
    watchEffect(()=>{
        const x1 = sum.value
        const x2 = person.age
        console.log('watchEffectй…ҚзҪ®зҡ„еӣһи°ғжү§иЎҢдәҶ')
    })

  • д»Җд№ҲжҳҜhookпјҹвҖ”вҖ” жң¬иҙЁжҳҜдёҖдёӘеҮҪж•°пјҢжҠҠsetupеҮҪж•°дёӯдҪҝз”Ёзҡ„Composition APIиҝӣиЎҢдәҶе°ҒиЈ…гҖӮ

  • зұ»дјјдәҺvue2.xдёӯзҡ„mixinгҖӮ

  • иҮӘе®ҡд№үhookзҡ„дјҳеҠҝ: еӨҚз”Ёд»Јз Ғ, и®©setupдёӯзҡ„йҖ»иҫ‘жӣҙжё…жҘҡжҳ“жҮӮгҖӮ

еҲӣе»әдёҖдёӘhookеҮҪж•°пјҢз”ЁеҲ°дәҶз»„еҗҲAPIпјҢеҚіеҸҜпјҢз„¶еҗҺеңЁйңҖиҰҒзҡ„ең°ж–№пјҢеңЁsetupйҮҢйқўеј•е…ҘеҚіеҸҜдҪҝз”ЁгҖӮ

import { reactive, onMounted, onBeforeUnmount } from 'vue'
export default function () {
  //е®һзҺ°йј ж ҮвҖңжү“зӮ№вҖқзӣёе…ізҡ„ж•°жҚ®
  let point = reactive({
    x: 0,
    y: 0
  })

  //е®һзҺ°йј ж ҮвҖңжү“зӮ№вҖқзӣёе…ізҡ„ж–№жі•
  function savePoint(event) {
    point.x = event.pageX
    point.y = event.pageY
    console.log(event.pageX, event.pageY)
  }

  //е®һзҺ°йј ж ҮвҖңжү“зӮ№вҖқзӣёе…ізҡ„з”ҹе‘Ҫе‘Ёжңҹй’©еӯҗ
  onMounted(() => {
    window.addEventListener('click', savePoint)
  })

  onBeforeUnmount(() => {
    window.removeEventListener('click', savePoint)
  })

  return point
}

shallowReactiveе’ҢshallowRefвҖӢ

  • shallowReactiveпјҡеҸӘеӨ„зҗҶеҜ№иұЎжңҖеӨ–еұӮеұһжҖ§зҡ„е“Қеә”ејҸпјҲжө…е“Қеә”ејҸпјүгҖӮеҸӘиҖғиҷ‘з¬¬дёҖеұӮж•°жҚ®зҡ„е“Қеә”ејҸгҖӮ

  • shallowRefпјҡеҸӘеӨ„зҗҶеҹәжң¬ж•°жҚ®зұ»еһӢзҡ„е“Қеә”ејҸ, дёҚиҝӣиЎҢеҜ№иұЎзҡ„е“Қеә”ејҸеӨ„зҗҶгҖӮеҰӮжһңжҳҜеҹәжң¬зұ»еһӢзҡ„иҜқпјҢshallowRefе’ҢRefжІЎд»Җд№ҲеҢәеҲ«гҖӮ

  • д»Җд№Ҳж—¶еҖҷдҪҝз”Ё?

    • еҰӮжһңжңүдёҖдёӘеҜ№иұЎж•°жҚ®пјҢз»“жһ„жҜ”иҫғж·ұ, дҪҶеҸҳеҢ–ж—¶еҸӘжҳҜеӨ–еұӮеұһжҖ§еҸҳеҢ– ===> shallowReactiveгҖӮ
    • еҰӮжһңжңүдёҖдёӘеҜ№иұЎж•°жҚ®пјҢеҗҺз»ӯеҠҹиғҪдёҚдјҡдҝ®ж”№иҜҘеҜ№иұЎдёӯзҡ„еұһжҖ§пјҢиҖҢжҳҜз”ҹж–°зҡ„еҜ№иұЎжқҘжӣҝжҚў ===> shallowRefгҖӮ

readonlyдёҺshallowReadonlyвҖӢ

  • readonly: и®©дёҖдёӘе“Қеә”ејҸж•°жҚ®еҸҳдёәеҸӘиҜ»зҡ„пјҲж·ұеҸӘиҜ»пјүгҖӮ
  • shallowReadonlyпјҡи®©дёҖдёӘе“Қеә”ејҸж•°жҚ®еҸҳдёәеҸӘиҜ»зҡ„пјҲжө…еҸӘиҜ»пјүгҖӮ
  • еә”з”ЁеңәжҷҜ: дёҚеёҢжңӣж•°жҚ®иў«дҝ®ж”№ж—¶гҖӮ жҜ”еҰӮеҲ«дәәеҶҷдәҶдёҖдёӘHookпјҢз”ЁreadonlyеҠ е·ҘеҗҺпјҢжҠҠж•°жҚ®еҜјеҮәпјҢиҝҷж ·еӯҗпјҢд»»дҪ•дәәз”Ёе°ұдёҚиғҪдҝ®ж”№гҖӮ

    • дҪҝз”ЁеңәжҷҜпјҡз”ЁдәҺиҜ»еҸ–е“Қеә”ејҸеҜ№иұЎеҜ№еә”зҡ„жҷ®йҖҡеҜ№иұЎпјҢеҜ№иҝҷдёӘжҷ®йҖҡеҜ№иұЎзҡ„жүҖжңүж“ҚдҪңпјҢдёҚдјҡеј•иө·йЎөйқўжӣҙж–°гҖӮ

    з”ЁtoRawеҢ…иЈ…еҗҺпјҢobjдјҡд»ҺProxyеҸҳжҲҗжӯЈеёёObjectеҜ№иұЎгҖӮ

customRefвҖӢ

  • дҪңз”ЁпјҡеҲӣе»әдёҖдёӘиҮӘе®ҡд№үзҡ„ refпјҢе№¶еҜ№е…¶дҫқиө–йЎ№и·ҹиёӘе’Ңжӣҙж–°и§ҰеҸ‘иҝӣиЎҢжҳҫејҸжҺ§еҲ¶гҖӮ

  • е®һзҺ°йҳІжҠ–ж•Ҳжһңпјҡ

    <template>
        <input type="text" v-model="keyword">
        <h3>{{keyword}}</h3>
    </template>

    <script>
        import {ref,customRef} from 'vue'
        export default {
            name:'Demo',
            setup(){
                // let keyword = ref('hello') //дҪҝз”ЁVueеҮҶеӨҮеҘҪзҡ„еҶ…зҪ®ref
                //иҮӘе®ҡд№үдёҖдёӘmyRef
                function myRef(value,delay){
                    let timer
                    //йҖҡиҝҮcustomRefеҺ»е®һзҺ°иҮӘе®ҡд№ү
                    return customRef((track,trigger)=>{
                        return{
                            get(){
                                track() //е‘ҠиҜүVueиҝҷдёӘvalueеҖјжҳҜйңҖиҰҒиў«вҖңиҝҪиёӘвҖқзҡ„
                                return value
                            },
                            set(newValue){
                                clearTimeout(timer)
                                timer = setTimeout(()=>{
                                    value = newValue
                                    trigger() //е‘ҠиҜүVueеҺ»жӣҙж–°з•Ңйқў
                                },delay)
                            }
                        }
                    })
                }
                let keyword = myRef('hello',500) //дҪҝз”ЁзЁӢеәҸе‘ҳиҮӘе®ҡд№үзҡ„ref
                return {
                    keyword
                }
            }
        }
    </script>
